#!/bin/sh -
################################################################################
# Uninstall apk from alpine linux.

docker_remove_apk() {
    # Remove apk
    apk del --repositories-file /dev/null apk-tools alpine-keys libc-utils

    # Delete apk installation data
    rm -rf /var/cache/apk /lib/apk /etc/apk
}
